 changeAll:function(e){
    if (e.detail.value == '全选'){
      this.setData({
        checkAll:true
      })
     for (var i in this.data.orderedItem){
       var temp_str = 'orderedItem[' + i + '].selected';
       if (!this.data.orderedItem[i].selected){
         this.setData({
          [temp_str]:true,
         })
       }
     }
    } else{
      this.setData({
        checkAll: false
      })
      for (var i in this.data.orderedItem) {
        var temp_str = 'orderedItem[' + i + '].selected';
        if (this.data.orderedItem[i].selected) {
          this.setData({
            [temp_str]: false,
          })
        }
      }
    }
  },
